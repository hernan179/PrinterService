package com.rta.opain.domain;

import java.io.Serializable;
import java.util.Date;
import javax.persistence.Basic;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.persistence.Temporal;
import javax.persistence.TemporalType;
import javax.xml.bind.annotation.XmlRootElement;

/**
 *
 * @author hsancheza
 */
@Entity
@Table(name = "cierres")
@XmlRootElement
public class Cierres implements Serializable {

    private static final long serialVersionUID = 1L;
    // @Max(value=?)  @Min(value=?)//if you know range of your decimal fields consider using these annotations to enforce field validation
    @Id
    @Basic(optional = false)
    @GeneratedValue(strategy = GenerationType.AUTO)
    @Column(name = "id_cierre")
    private Integer idCierre;

    @Column(name = "fac_inicio")
    String facInicio;
    @Column(name = "fac_fin")
    String facFin;
    @Column(name = "fecha")
    @Temporal(TemporalType.TIMESTAMP)
    Date fecha;
    @Column(name = "valor_total")
    Integer valorTotal;

    public Integer getIdCierre() {
        return idCierre;
    }

    public void setIdCierre(Integer idCierre) {
        this.idCierre = idCierre;
    }

    public String getFacInicio() {
        return facInicio;
    }

    public void setFacInicio(String facInicio) {
        this.facInicio = facInicio;
    }

    public String getFacFin() {
        return facFin;
    }

    public void setFacFin(String facFin) {
        this.facFin = facFin;
    }

    public Date getFecha() {
        return fecha;
    }

    public void setFecha(Date fecha) {
        this.fecha = fecha;
    }

    public Integer getValorTotal() {
        return valorTotal;
    }

    public void setValorTotal(Integer valorTotal) {
        this.valorTotal = valorTotal;
    }

    public Integer getCantidadFac() {
        return cantidadFac;
    }

    public void setCantidadFac(Integer cantidadFac) {
        this.cantidadFac = cantidadFac;
    }

    public Integer getCantidadAnulaciones() {
        return cantidadAnulaciones;
    }

    public void setCantidadAnulaciones(Integer cantidadAnulaciones) {
        this.cantidadAnulaciones = cantidadAnulaciones;
    }

    public Integer getTotalE() {
        return totalE;
    }

    public void setTotalE(Integer totalE) {
        this.totalE = totalE;
    }

    public Integer getTotalD() {
        return totalD;
    }

    public void setTotalD(Integer totalD) {
        this.totalD = totalD;
    }

    public Integer getTotalC() {
        return totalC;
    }

    public void setTotalC(Integer totalC) {
        this.totalC = totalC;
    }

    public String getUsuario() {
        return usuario;
    }

    public void setUsuario(String usuario) {
        this.usuario = usuario;
    }

    @Column(name = "cantidad_fac")
    Integer cantidadFac;
    @Column(name = "cantidad_anulaciones")
    Integer cantidadAnulaciones;

    @Column(name = "total_e")
    Integer totalE;

    @Column(name = "total_d")
    Integer totalD;
    @Column(name = "total_c")
    Integer totalC;

    @Column(name = "usuario")
    String usuario;

}
